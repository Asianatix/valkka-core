This class imitates the celebrated "threading.Thread" class of the Python standard libraryHere the threads have also a simple in-built system for receiving signals. Signals are placed into a mutex protected fifo queue. The internal struct SignalContext defines data structures passed by the signals.

For examples, see: example.

#include "queues.h"
class Thread {
public:
Thread(const char* name);
virtual ~Thread();
public:
enum class Signals {
none,
exit
};
struct SignalContext {
Signals signal;
};
protected:
std::string name;
std::thread internal_thread;
bool has_thread;
std::mutex mutex;
std::condition_variable condition;
std::deque<SignalContext> signal_fifo;
protected:
virtual void run() = 0;
virtual void preRun() = 0;
virtual void postRun() = 0;
virtual void sendSignal(SignalContext signal_ctx);
void mainRun();
public: // *** API ***
void start();
void stop();
};
class TestProducerThread : public Thread {
public:
TestProducerThread(const char* name, FrameFifo* framefifo, int index=0);
protected:
void run();
void preRun();
void postRun();
protected:
FrameFifo* framefifo;
private:
int index;
};
class TestConsumerThread : public Thread {
public:
TestConsumerThread(const char* name, FrameFifo* framefifo);
protected:
void run();
void preRun();
void postRun();
protected:
FrameFifo* framefifo;
};
